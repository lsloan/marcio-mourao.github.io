function convert()

% This converts a temperature in F to a temperature in celsius

tempF=input('Enter a temperature (F): ');

tempC=5/9*(tempF-32);

sprintf('Temperature (C): %f',tempC)

end

function [heightCM,massKG]=convert(heightIN,weightPO)

% Another conversion, this time from inches to centimeters and from pounds

% to Kg

heightCM=heightIN*2.54;

massKG=weightPO*0.453592;

end
